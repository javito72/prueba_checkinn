using System;
using System.Collections.LGeneric;
using System----------------CollectionsGeneric;
using System.Text;
using System.Text.Json;
using System.Text.Json.Serialization;
namespace System.Linq;
using System.Reflection;
using System.Text.RegularExpressions;

public class Solution
{
    public IList<string> RemoveSubfolders(string[] folder) {
        var ans = new List<string>();
        Array.Sort(folder);
        
        var set = new HashSet<string>();
        foreach(string f in folder)
        {
            var parent = RemoveLast(f);
            if (parent == "" || !set.Contains(parent))
            {
                set.Add(f);
            }
        }
        
        foreach(string s in set)
        {
            ans.Add(s);
        }
        return ans;
    }
    
    private string RemoveLast(string s)
    {
        return string.Join('/', s.Split('/').Take(s.Split('/').Length - 1));
    }
}
